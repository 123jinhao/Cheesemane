    <script>
        const commons = ["flour", "water", "milk", "egg", "knife", "chili", "dough", "cheese", "tomato", "potato", "sliced mushroom", "spring onion", "apple", "pepper", "salt", "sugar", "fire", "pepperoni", "cherry", "oil", "pinecone", "pineapple"];
        const rares = ["a bar of chocolate", "cream", "carrot", "cabbage", "pumpkin", "onion", "corn", "unknown fish", "tuna", "salmon", "almond", "peanut", "grape", "grate", "honey", "cactus", "sesame seed", "seaweed", "kimchi(spicy)", "mustard", "oat", "cocoa powder", "sour powder", "aijinomoto", "lime", "chicken(raw)", "banana", "strawberry", "butter", "maple syrup", "acorn", "orange", "kimchi(spiceless)", "mayonese"];

        let score = 0;
        let currentToppings = [];
        let currentRequirement = "";
        let currentCustomer = "";

        const customers = [
            { name: "The Grumpy Woodsman", want: "pinecone", dialog: "I want something that reminds me of the forest!" },
            { name: "Fancy Feline", want: "tuna", dialog: "Meow! I require the finest fish you have." },
            { name: "The Spicy Grandma", want: "kimchi(spicy)", dialog: "Make it burn! I need some heat!" },
            { name: "The Sweet Tooth", want: "honey", dialog: "Make it sticky and sweet, please!" },
            { name: "The Weird Scientist", want: "unknown fish", dialog: "I'm conducting a flavor experiment. Give me something... mysterious." },
            { name: "The Traditionalist", want: "pepperoni", dialog: "Just a classic pepperoni for me." }
        ];

        function init() {
            const commonShelf = document.getElementById('common-shelf');
            const rareShelf = document.getElementById('rare-shelf');

            commons.forEach(item => {
                let div = document.createElement('div');
                div.className = 'ingredient';
                div.innerText = item;
                div.onclick = () => { currentToppings.push(item); updateUI(); };
                commonShelf.appendChild(div);
            });

            rares.forEach(item => {
                let div = document.createElement('div');
                div.className = 'ingredient rare';
                div.innerText = item;
                div.onclick = () => { currentToppings.push(item); updateUI(); };
                rareShelf.appendChild(div);
            });
        }

        function nextCustomer() {
            let rand = customers[Math.floor(Math.random() * customers.length)];
            currentCustomer = rand.name;
            currentRequirement = rand.want;
            document.getElementById('customer-name').innerText = "CUSTOMER: " + currentCustomer;
            document.getElementById('customer-bubble').innerText = `"${rand.dialog}"`;
            resetPizza();
        }

        function updateUI() {
            document.getElementById('topping-list').innerText = currentToppings.length > 0 ? currentToppings.join(", ") : "Empty Pan";
            document.getElementById('score').innerText = score;
        }

        function bakePizza() {
            if (!currentCustomer) { alert("No customer to serve!"); return; }
            if (!currentToppings.includes("fire")) { alert("It's raw! You need 'fire'!"); return; }

            if (currentToppings.includes(currentRequirement)) {
                score += 100;
                alert(currentCustomer + " says: PERFECT! Exactly what I wanted!");
            } else {
                score -= 50;
                alert(currentCustomer + " says: Eww! This isn't what I asked for!");
            }
            nextCustomer();
        }

        function resetPizza() {
            currentToppings = [];
            updateUI();
        }

        init();
    </script>
